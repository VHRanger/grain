1 foreach (i; [2, 3, 4]) {
2     foreach (j; [2, 3, 4]) {

5         import mir.ndslice : slice;
6         import grain.testing;
7 
8         auto k = 3;
9         auto a = uniform!float(i, k).slice.variable;
10         auto b = uniform!float(k, j).slice.variable;
11         auto gc = uniform!float(i, j).slice.variable;
12         MatMul!float func;
13         gradCheck(func, tuple(a, b), gc, 1e-3, 1e-3, 1e-3);
14 
15         version (grain_cuda) {
16             import numir.testing;

20             assert(approxEqual(dc.to!HostStorage.sliced, hc.sliced));
21             auto hgab = func.backward(gc);

matmul with variable.backward

1 import std.typecons;
2 import numir;
3 import mir.ndslice;
4 grain.autograd.backprop = true;
5 auto func = new MatMul!float;
6 auto a = uniform!float(3, 4).slice.variable(true);
7 auto b = uniform!float(4, 2).slice.variable(true);
8 auto c = func.applyForward(a, b);
9 auto gc = uniform!float(3, 2).slice.variable;
10 auto ugc = UntypedVariable(gc);
11 c.backward(&ugc);
12 auto gab = func.backward(gc);
13 assert(a.gradSlice == gab[0].sliced);
14 assert(b.gradSlice == gab[1].sliced);
