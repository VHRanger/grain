12 /// enumerate the parameter names inside chain C
13 enum variableNames(C) = {
14     string[] ret;
15     void register(V)(string k, V v) if (isVariable!V) {
16         ret ~= [k];
17     }
18 
19     C chain;
20     iterVariables!((k, v) { register(k, v); })(&chain, "");
21     return ret;

53             this.fc2 = L(nhidden, nhidden);
54             this.fc3 = L(nhidden, 10);
55         }

57         auto opCall(Variable!(T, 2, Storage) x) {
58             auto h1 = relu(this.fc1(x));

60             auto h3 = this.fc3(h2);

66 /// convert D type into HDF5 type-id https://support.hdfgroup.org/HDF5/doc1.8/RM/PredefDTypes.html
67 auto toH5Type(T)() {
68     import std.traits;
69     import std.format;
70 
71     static assert(isBasicType!T);
72     mixin("return H5T_%s%dLE;".format(isFloatingPoint!T ? "IEEE_F"
73             : (isSigned!T ? "STD_I" : "STD_U"), T.sizeof * 8));
74 }

76 /// save chain parameters to HDF5 path
77 void save(bool verbose = true, C)(C chain, string path) {
78     import std.file : exists;
79     import std..string : replace, endsWith;
80     import mir.ndslice : slice;
81     import grain.utility : castArray;
82 
83     auto file = H5Fcreate(path.toStringz, // path.exists ? H5F_ACC_TRUNC :
84             H5F_ACC_TRUNC, H5P_DEFAULT, H5P_DEFAULT);
85     scope (exit)
86         H5Fclose(file);
87 
88     void register(T, size_t dim, alias Storage)(string k, Variable!(T, dim, Storage) v) {
89         auto h = v.to!HostStorage;
90         // FIXME support check contiguous
91         // auto s = h.sliced.slice;
92         auto data = v.to!HostStorage.data;
93         auto dims = h.shape.castArray!hsize_t;
94         auto space = H5Screate_simple(cast(int) dims.length, dims.ptr, dims.ptr);
95         scope (exit)
96             H5Sclose(space);
97         auto dataset = H5Dcreate2(file, toStringz("/" ~ k), toH5Type!T, space,
98                 H5P_DEFAULT, H5P_DEFAULT, H5P_DEFAULT);
99         scope (exit)
100             H5Dclose(dataset);
101         H5Dwrite(dataset, toH5Type!T, H5S_ALL, H5S_ALL, H5P_DEFAULT, cast(void*) data
102                 .ptr);
103     }
104 
105     iterVariables!((k, v) { register(k, v); })(&chain, "");
106 }

108 /// load chain parameters from HDF5 path
109 void load(C)(ref C chain, string path) {
110     import std..string : replace, endsWith;
111     import mir.ndslice : slice, sliced;
112 
113     import grain.utility : castArray;
114 
115     auto file = H5Fopen(path.toStringz, // path.exists ? H5F_ACC_RDWR :
116             H5F_ACC_RDONLY, //
117             // H5F_ACC_RDWR,
118             H5P_DEFAULT);
119     scope (exit)
120         H5Fclose(file);
121 
122     void register(T, size_t dim, alias Storage)(string k, ref Variable!(T, dim, Storage) v) {
123         auto dataset = H5Dopen2(file, toStringz("/" ~ k), H5P_DEFAULT);
124         scope (exit)
125             H5Dclose(dataset);
126 
127         auto raw = new T[v.data.length];
128         H5Dread(dataset, toH5Type!T, H5S_ALL, H5S_ALL, H5P_DEFAULT, cast(void*)&raw[
129                 0]);
130 
131         auto src = raw.sliced(v.shape.castArray!size_t).variable;
132         static if (is(Storage!T == HostStorage!T)) {
133             v.sliced[] = src.sliced;
134         }
135         else {
136             import grain.cudnn : transform;
137 
138             transform(src.to!Storage, v);
139         }
140     }
141 
142     refIterVariables!((k, ref v) { register(k, v); })(chain, "");
143 }
