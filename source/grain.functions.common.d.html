18 /// a simple type check of forward/backward functions compatibility
19 mixin template TypeChecker(alias forward, alias backward) {
20     static assert(allSatisfy!(isVariable, Parameters!forward),
21                   "all the forward function args should be variable.");
22     static assert(allSatisfy!(isVariable, Parameters!backward),
23                   "all the backward function args should be variable.");
24     static if (arity!forward == 1 && arity!backward == 1) {
25         static assert(is(ReturnType!backward == Parameters!forward[0]));
26         static assert(is(ReturnType!forward == Parameters!backward[0]));
27     } else static if (arity!backward == 1) {
28         static assert(is(ReturnType!backward == Tuple!(Parameters!forward)));
29         static assert(is(ReturnType!forward == Parameters!backward[0]));
30     } else static if (arity!forward == 1) {
31         static assert(is(ReturnType!backward == Parameters!forward[0]));
32         static assert(is(ReturnType!forward == Tuple!(Parameters!backward)));
33     } else {
34         static assert(is(ReturnType!backward == Tuple!(Parameters!forward)));
35         static assert(is(ReturnType!forward == Tuple!(Parameters!backward)));
36     }
37 }

61 /// common components (typecheck and backprop wrappers) for autograd functions
62 mixin template FunctionCommon() {
63     import std.meta : allSatisfy;
64     import std.typecons : isTuple, tuple, Tuple, RefCounted;
65     import std.traits : arity, Parameters, ReturnType;
66 
67     @disable this(this); // no copyable
68 
69     static foreach (i, forward; __traits(getOverloads, typeof(this), "forward")) {
70         static foreach (i, backward; __traits(getOverloads, typeof(this), "backward")) {
71             static if (!allSatisfy!(isHost, Parameters!forward) &&
72                        !allSatisfy!(isHost, Parameters!backward)) {
73                 alias DeviceRets = Tuple!(Parameters!backward);
74                 alias DeviceArgs = Tuple!(Parameters!forward);
75                 mixin TypeChecker!(forward, backward);
76             }
77             static if (allSatisfy!(isHost, Parameters!forward) &&
78                        allSatisfy!(isHost, Parameters!backward)) {
79                 alias HostRets = Tuple!(Parameters!backward);
80                 alias HostArgs = Tuple!(Parameters!forward);
81                 mixin TypeChecker!(forward, backward);
82             }
83         }
84     }
85     static assert(isFunction!(typeof(this)));
86 
87     /// store grain.autograd.BackProp object in returned variables from forward function
88     auto applyForward(Args...)(Args args) {
89         import std.algorithm : each;
90         RefCounted!(UntypedVariable[]) uargs;
91         uargs.length = args.length;
92         foreach (i, a; args) {
93             uargs[i] = UntypedVariable(a);
94             uargs[i].bprop = a.bprop; // pass the chain to backprop
95         }
96         auto rets = this.forward(args).toTuple;
97         enum isHost = allSatisfy!(isHost, Args);
98         foreach (i, r; rets) {
99             auto u = UntypedVariable(r);
100             if (grain.autograd.backprop) {
101                 // RefCounted!
102                 BackProp bp = BackProp(&this.applyBackward!isHost,
103                                        uargs);
104                 bp.gradOutputs.length = rets.length;
105                 u.bprop = bp;
106                 u.outPosition = i;
107                 rets[i].bprop = bp;
108             }
109         }
110         static if (rets.length > 1) {
111             return rets;
112         } else {
113             return rets[0];
114         }
115     }

208 /// check if broadcastable
209 auto broadcastable(T, size_t dim, alias Storage)(Variable!(T, dim, Storage) a, Variable!(T, dim, Storage) b) {
210     import std.typecons : tuple;
211     import std.algorithm : max;
212     int[dim] resultShape;
213     bool ok = false;
214     foreach (i; 0 .. dim) {
215         ok = a.shape[i] == b.shape[i] || a.shape[i] == 1 || b.shape[i] == 1;
216         if (ok) {
217             resultShape[i] = max(a.shape[i], b.shape[i]);
218         } else break;
219     }
220     return tuple!("ok", "shape")(ok, resultShape);
221 }

223 /// expand dimension i.e. repeat n time on dim
224 auto expand(size_t dim, S)(S s, size_t n) if (isSlice!S) {
225     import mir.primitives : DimensionCount;
226     static assert(dim < DimensionCount!S, format!"acessing invalid dim %d (should be < %d)"(dim, Ndim!S));
227     assert(s.length!dim == 1);
228 
229     import mir.ndslice : repeat, swapped, transposed, unpack;
230     /// [a, 1, b] -> repeat [n, a, 1, b] -> swapped [1, a, n, b]
231     return s.repeat(n).unpack.swapped!(0, dim+1)[0];
232 }

246 /// exapand dimension if s.length!dim == 1 else do nothing but type in the same expressions of repeat/unpack/swapped/index[0]
247 auto maybeExpand(size_t dim, S)(S s, size_t n) if (isSlice!S) {
248     import mir.ndslice;
249     import mir.ndslice : repeat, swapped, transposed, unpack;
250     return s.length!dim == 1 ? s.expand!dim(n) :
251         /// [a, c, b] -> repeat [1, a, c, b] -> swapped [1, a, c, b]
252         s.repeat(1).unpack.swapped!(dim+1, dim+1)[0];
253 }

263 /**
264    Returns:
265       broadcasted slice.
266       For example, when a has its shape [a, 1] and b has [1, b],
267       this function returns expanded a and b with a broadcasted shape [a, b].
268 
269    See_also:
270       https://docs.scipy.org/doc/numpy-1.13.0/user/basics.broadcasting.html
271 */
272 auto broadcast(S1, S2)(S1 a0, S2 b0) if (isSlice!S1 && isSlice!S2) {
273     import std.format : format;
274     import std.typecons : tuple;
275     import mir.primitives : DimensionCount;
276     static assert(DimensionCount!S1 == DimensionCount!S2); // TODO support dim mismatched slices by unsqueezing like numpy
277     enum dim = DimensionCount!S1;
278     static foreach (d; 1 .. dim+1) {
279         mixin(format!q{auto a%d = a%d.maybeExpand!(d-1)(b0.length!(d-1));}(d, d-1));
280         mixin(format!q{auto b%d = b%d.maybeExpand!(d-1)(a0.length!(d-1));}(d, d-1));
281     }
282     mixin(format!q{auto ax = a%d;}(dim));
283     mixin(format!q{auto bx = b%d;}(dim));
284     return tuple(ax, bx);
285 }

298 /// reduce slice into targetShape, TODO @nogc
299 auto reduceShape(alias fun, S, size_t N)(S s0, size_t[N] targetShape) {
300     import numir;
301     import mir.ndslice;
302     import mir.math : sum;
303     import std.format : format;
304     import std.exception : assumeWontThrow; // TODO unsqueeze can be assumeWontThrow
305 
306     static if (N == 1) {
307         return s0;
308     } else {
309         auto rec(size_t n, T)(T t) {
310             static if (n == N) return t;
311             else {
312                 return
313                     rec!(n+1)(
314                         targetShape[n] == 1
315                         ? assumeWontThrow(t.alongDim!n.map!fun.slice.unsqueeze!n).slice
316                         : t.slice);
317             }
318         }
319         return rec!0(s0);
320     }
321 }
