22 /// c = op(alpha1 * a + alpha2 * b) + beta * c;
23 struct OpBinary(T, size_t dim, string ops) if (isFloatingPoint!T) {
24     import mir.ndslice;
25 
26     T alpha1 = 1, alpha2 = 1;
27 
28     uint[dim] shape1, shape2;
29     Variable!(T, dim, HostStorage) ha, hb;
30 
31     auto forward(Variable!(T, dim, HostStorage) a, Variable!(T, dim, HostStorage) b) {
32         auto info = broadcastable(a, b);
33         assert(info.ok);
34         auto abx = broadcast(a.sliced, b.sliced);
35         auto ax = abx[0];
36         auto bx = abx[1];
37         auto c = slice(this.alpha1 * ax);
38 
39         // TODO if train
40         this.shape1 = a.shape;
41         this.shape2 = b.shape;
42         // ops
43         static if (ops == "+") {
44             c[] += this.alpha2 * bx;
45         } else static if (ops == "*") {
46             this.ha = ax.variable;
47             this.hb = bx.variable;
48             c[] *= this.alpha2 * bx;
49         } else {
50             static assert("unknown operator: " ~ ops);
51         }
52         return c.variable(a.requiresGrad || b.requiresGrad);
53     }
54 
55     auto backward(Variable!(T, dim, HostStorage) gc) {
56         import numir;
57         import mir.math : sum;
58         import mir.ndslice;
59         static if (ops == "+") {
60             auto ga = this.alpha1 == 1 ? gc.sliced.slice.universal : slice(this.alpha1 * gc.sliced).universal;
61             if (ga.shape != this.shape1) {
62                 ga = reduceShape!(sum!"fast")(ga, this.shape1.castArray!size_t).universal;
63             }
64             auto gb = this.alpha2 == 1 ? gc.sliced.slice.universal : slice(this.alpha2 * gc.sliced).universal;
65             if (gb.shape != this.shape2) {
66                 gb = reduceShape!(sum!"fast")(gb, this.shape2.castArray!size_t).universal;
67             }
68             return tuple(ga.variable, gb.variable);
69         } else static if (ops == "*") {
70             assert(this.ha.defined);
71             assert(this.hb.defined);
72             auto ga = gc.sliced.slice.universal;
73             ga[] *= this.alpha1 * this.alpha2 * this.hb.sliced;
74             if (ga.shape != this.shape1) {
75                 ga = reduceShape!(sum!"fast")(ga, this.shape1.castArray!size_t).universal;
76             }
77             auto gb = gc.sliced.slice.universal;
78             gb[] *= this.alpha1 * this.alpha2 * this.ha.sliced;
79             if (gb.shape != this.shape2) {
80                 gb = reduceShape!(sum!"fast")(gb, this.shape2.castArray!size_t).universal;
81             }
82             return tuple(ga.variable, gb.variable);
83         } else {
84             static assert("unknown operator: " ~ ops);
85         }
86     }
87 
88     version (grain_cuda) {
89         import grain.cudnn;
90         import derelict.cudnn7;
91         import std.algorithm : find;
92 
93         enum opBinaryDict = [
94                              "+": CUDNN_OP_TENSOR_ADD,
95                              "*": CUDNN_OP_TENSOR_MUL,
96                              "min": CUDNN_OP_TENSOR_MIN,
97                              "max": CUDNN_OP_TENSOR_MAX
98                              ];
99 
100         static if (opBinaryDict.keys.find(ops)) {
101             static if (ops == "*") {
102                 Variable!(T, dim, DeviceStorage) da, db;
103             }
104 
105             auto forward(Variable!(T, dim, DeviceStorage) a, Variable!(T, dim, DeviceStorage) b) {
106                 // TODO implement non-cudnn case
107                 foreach (d; 0 .. dim) {
108                     assert(a.shape[d] == b.shape[d] || b.shape[d] == 1,
109                            "cuDNN does not support complete broadcasting");
110                 }
111                 // TODO if train
112                 this.shape1 = a.shape;
113                 this.shape2 = b.shape;
114                 static if (ops == "*") {
115                     this.da = a;
116                     this.db = b;
117                 }
118 
119                 auto c = a.uninit;
120                 c.requiresGrad = a.requiresGrad || b.requiresGrad;
121                 import grain.cudnn;
122                 tensorOp!(opBinaryDict[ops], T, dim)(c, a, b, this.alpha1, this.alpha2);
123                 return c;
124             }
125         } else {
126             static assert("unknown operator: " ~ ops);
127         }
128 
129         static if (ops == "+") {
130             auto backward(Variable!(T, dim, DeviceStorage) gc) {
131                 Variable!(T, dim, DeviceStorage) ga, gb;
132                 if (this.shape1 == gc.shape) {
133                     ga = gc.dup;
134                     if (this.alpha1 != 1.0) grain.cudnn.scale(ga, this.alpha1);
135                 } else {
136                     ga = uninitVariable!(T, DeviceStorage)(this.shape1);
137                     grain.cudnn.reduce!CUDNN_REDUCE_TENSOR_ADD(gc, ga, this.alpha1);
138                 }
139 
140                 if (this.shape2 == gc.shape) {
141                     gb = gc.dup;
142                     if (this.alpha2 != 1.0) grain.cudnn.scale(gb, this.alpha2);
143                 } else {
144                     gb = uninitVariable!(T, DeviceStorage)(this.shape2);
145                     grain.cudnn.reduce!CUDNN_REDUCE_TENSOR_ADD(gc, gb, this.alpha2);
146                 }
147                 return tuple(ga, gb);
148             }
149         } else static if (ops == "*") {
150             auto backward(Variable!(T, dim, DeviceStorage) gc) {
151                 auto gax = uninitVariable!(T, DeviceStorage)(gc.shape);
152                 auto gbx = uninitVariable!(T, DeviceStorage)(gc.shape);
153                 auto alpha = this.alpha1 * this.alpha2;
154                 grain.cudnn.tensorOp!CUDNN_OP_TENSOR_MUL(gax, gc, this.db, alpha);
155                 grain.cudnn.tensorOp!CUDNN_OP_TENSOR_MUL(gbx, gc, this.da, alpha);
156 
157                 Variable!(T, dim, DeviceStorage) ga, gb;
158                 if (this.shape1 == gc.shape) {
159                     ga = gax;
160                 } else {
161                     ga = uninitVariable!(T, DeviceStorage)(this.shape1);
162                     grain.cudnn.reduce!CUDNN_REDUCE_TENSOR_ADD(gax, ga);
163                 }
164 
165                 if (this.shape2 == gc.shape) {
166                     gb = gbx;
167                 } else {
168                     gb = uninitVariable!(T, DeviceStorage)(this.shape2);
169                     grain.cudnn.reduce!CUDNN_REDUCE_TENSOR_ADD(gbx, gb);
170                 }
171                 return tuple(ga, gb);
172             }
173         }
174     }
175 
176     mixin FunctionCommon;
177 }

278 /++
279  Matrix-Matrix multiplication (using cuBLAS)
280 
281  See_Also: https://github.com/chainer/chainer/blob/v1/chainer/functions/connection/linear.py#L11
282  +/
283 struct MatMul(T) {
284     import mir.ndslice : transposed, universal;
285     import std.typecons : tuple;
286     import lubeck : mtimes;
287     T alpha = 1;
288     T beta = 0;
289     Variable!(T, 2, HostStorage) ha, hb;
290 
291     // TODO uncomment this line
292     mixin FunctionCommon;
293 
294     auto forward(Variable!(T, 2, HostStorage) a, Variable!(T, 2, HostStorage) b) {
295         // TODO if training
296         this.ha = a;
297         this.hb = b;
298         return mtimes(a.sliced, b.sliced).variable(a.requiresGrad || b.requiresGrad);
299     }
300 
301     auto backward(Variable!(T, 2, HostStorage) gy) {
302         auto ga = mtimes(gy.sliced, this.hb.sliced.transposed).variable;
303         auto gb = mtimes(this.ha.sliced.transposed, gy.sliced).variable;
304         return tuple(ga, gb);
305     }
306 
307     version(grain_cuda) {
308         Variable!(T, 2, DeviceStorage) da, db;
309 
310         auto forward(Variable!(T, 2, DeviceStorage) a, Variable!(T, 2, DeviceStorage) b) {
311             import grain.cublas;
312             static if (is(T == float)) {
313                 alias gemm = cublasSgemm_v2;
314             } else static if (is(T == double)) {
315                 alias gemm = cublasDgemm_v2;
316             } else {
317                 static assert(false, "unsupported type");
318             }
319 
320             import std.typecons : RefCounted;
321             assert(a.shape[1] == b.shape[0]);
322             auto cdata = CuArray!T(a.shape[0] * b.shape[1]);
323             auto c = Variable!(T, 2, DeviceStorage)(
324                 a.requiresGrad || b.requiresGrad,
325                 [a.shape[0], b.shape[1]], [b.shape[1], 1], cdata);
326             // C = A x B = (BT x AT)T
327             // TODO support transposed (CUBLAS_OP_T)
328             // see https://github.com/libmir/mir-blas/blob/master/source/mir/blas.d#L299
329             // TODO if train
330             this.da = a;
331             this.db = b;
332             checkCublasErrors(gemm(cublasHandle, CUBLAS_OP_N, CUBLAS_OP_N,
333                                    cast(int) b.shape[1],
334                                    cast(int) a.shape[0], cast(int) a.shape[1],
335                                    &alpha,
336                                    cast(const T*) b.data.ptr, cast(int) b.strides[0],
337                                    cast(const T*) a.data.ptr, cast(int) a.strides[0],
338                                    &beta,
339                                    cast(T*) c.data.ptr, cast(int) c.strides[0]));
340             return c;
341         }
342 
343         auto backward(Variable!(T, 2, DeviceStorage) gc) {
344             import grain.cublas;
345             static if (is(T == float)) {
346                 alias gemm = cublasSgemm_v2;
347             } else static if (is(T == double)) {
348                 alias gemm = cublasDgemm_v2;
349             } else {
350                 static assert(false, "unsupported type");
351             }
352             auto ga = this.da.dup;
353             auto gb = this.db.dup;
354             // auto ga = mtimes(gc.sliced, this.hb.sliced.transposed).variable;
355             checkCublasErrors(gemm(cublasHandle, CUBLAS_OP_T, CUBLAS_OP_N,
356                                    cast(int) ga.shape[1],
357                                    cast(int) ga.shape[0], cast(int) gc.shape[1],
358                                    &alpha,
359                                    cast(const T*) db.data.ptr, cast(int) db.strides[0],
360                                    cast(const T*) gc.data.ptr, cast(int) gc.strides[0],
361                                    &beta,
362                                    cast(T*) ga.data.ptr, cast(int) ga.strides[0]));
363             // auto gb = mtimes(this.ha.sliced.transposed, gc.sliced).variable;
364             checkCublasErrors(gemm(cublasHandle, CUBLAS_OP_N, CUBLAS_OP_T,
365                                    cast(int) gb.shape[1],
366                                    cast(int) gb.shape[0], cast(int) da.shape[0],
367                                    &alpha,
368                                    cast(const T*) gc.data.ptr, cast(int) gc.strides[0],
369                                    cast(const T*) da.data.ptr, cast(int) da.strides[0],
370                                    &beta,
371                                    cast(T*) gb.data.ptr, cast(int) gb.strides[0]));
372             return tuple(ga, gb);
373         }
374     }
375 }

427 /**
428    Add bias vector to matrix used inside grain.chain.Linear
429    TODO: generalize to broadcastable addition (use cudnnAddTensor)
430 */
431 struct AddBias(T) {
432     mixin FunctionCommon;
433 
434     import mir.ndslice : map, slice;
435     import std.typecons : tuple;
436     auto forward(Variable!(T, 2, HostStorage) a, Variable!(T, 1, HostStorage) b) {
437         assert(a.shape[1] == b.shape[0]);
438         auto ret = a.dup;
439         foreach (i; 0 .. a.shape[0]) {
440             ret.sliced[i][] += b.sliced;
441         }
442         return ret;
443     }
444 
445     auto backward(Variable!(T, 2, HostStorage) gy) {
446         import numir : alongDim;
447         import mir.math : sum;
448         auto gb = gy.sliced.alongDim!0.map!sum.slice.variable;
449         return tuple(gy, gb);
450     }
451 
452     version (grain_cuda) {
453         import grain.kernel : addBias, addBiasGrad;
454 
455         auto forward(Variable!(T, 2, DeviceStorage) a, Variable!(T, 1, DeviceStorage) b) {
456             assert(a.shape[1] == b.shape[0]);
457             auto y = a.dup;
458             auto n = cast(uint) y.data.length;
459             auto blen = cast(uint) b.data.length;
460             Global.kernel!addBias
461                 .call(y.data.ptr, b.data.ptr, blen, n).launch(n);
462             return y;
463         }
464 
465         auto backward(Variable!(T, 2, DeviceStorage) gy) {
466             auto gb = CuArray!T(gy.shape[1]);
467             gb.zero_();
468             auto n = cast(uint) gy.data.length;
469             auto blen = cast(uint) gb.length;
470             Global.kernel!addBiasGrad
471                 .call(gy.data.ptr, gb.ptr, blen, n).launch(n);
472             return tuple(gy, Variable!(T, 1, DeviceStorage)(false, [cast(int) blen], [1], gb));
473         }
474     }
475 }

509 /// Emebedding ID into vector. TODO: support N-dim input. support sparse weight matrix
510 struct Embedding(T) {
511     import std.range : enumerate;
512     import numir : view, empty, zeros;
513 
514     Variable!(int, 1, HostStorage) hx;
515     uint[2] wshape;
516 
517     auto forward(Variable!(T, 2, HostStorage) weight, Variable!(int, 1, HostStorage) ids) {
518         this.hx = ids; // TODO if train
519         this.wshape = weight.shape; // TODO if train
520         auto ys = empty!T(ids.shape[0], weight.shape[1]);
521         foreach (i, id; ids.sliced.enumerate) {
522             ys[i, 0..$] = weight.sliced[id, 0..$];
523         }
524         return ys.variable(weight.requiresGrad);
525     }
526 
527     auto backward(Variable!(T, 2, HostStorage) gy) {
528         auto gw = zeros!T(this.wshape.castArray!size_t);
529         foreach (i, id; this.hx.sliced.enumerate) {
530             gw[id, 0..$] += gy.sliced[i];
531         }
532         return tuple(gw.variable(gy.requiresGrad), typeof(this.hx)());
533     }
534 
535     version (grain_cuda) {
536         Variable!(int, 1, DeviceStorage) dx;
537 
538         auto forward(Variable!(T, 2, DeviceStorage) weight, Variable!(int, 1, DeviceStorage) ids) {
539             import grain.kernel : embedding;
540             this.dx = ids; // TODO if train
541             this.wshape = weight.shape; // TODO if train
542             auto ys = uninitVariable!(T, DeviceStorage, 2)([ids.shape[0], weight.shape[1]], weight.requiresGrad);
543             Global.kernel!embedding
544                 .call(weight.data.ptr, ids.data.ptr, ys.data.ptr, weight.shape[0], weight.shape[1], ids.shape[0])
545                 .launch(weight.shape[1] * ids.shape[0]);
546             return ys;
547         }
548 
549         auto backward(Variable!(T, 2, DeviceStorage) gy) {
550             import grain.kernel : embeddingGrad;
551             auto gw = uninitVariable!(T, DeviceStorage, 2)(this.wshape, gy.requiresGrad);

586 void generateStrides(const int* dimA, int* strideA, int nbDims, bool isNchw) {
587     if (isNchw) {
588         strideA[nbDims-1] = 1 ;
589         for(int d = nbDims-2 ; d >= 0 ; d--) {
590             strideA[d] = strideA[d+1] * dimA[d+1] ;
591         }
592     } else {
593         strideA[1] = 1;
594         strideA[nbDims-1] = strideA[1]*dimA[1];
595         for(int d = nbDims-2 ; d >= 2 ; d--) {
596             strideA[d] = strideA[d+1] * dimA[d+1] ;
597         }
598         strideA[0] = strideA[2]*dimA[2];
599     }
600 }

602 /** Convert a linear index
603 i = d_1 s_1 ... s_n + d_2 s_2 ... s_n + d_n-1 s_n + d_n
604 into a multidimensional index
605 (d_1, d_2, ..., d_n)
606 */
607 void lin2dim(size_t length)(int id, scope ref int[length] ids, const ref int[length] dims) {
608     int idrem = id ;
609     int prod  = 1 ; // accumulates the product of the dimensions
610     foreach_reverse(i; 0 .. length) {
611         ids[i] = (idrem / prod) % dims[i] ;
612         idrem = id - ids[i] * prod ;
613         prod *= dims[i] ;
614     }
615 }
616 
617 void doEpilog(float[] o, int idx, float alphaAcc, float beta) {
618     if( beta == 0f ) {
619         o[idx] = alphaAcc;
620     } else {
621         o[idx] = alphaAcc + o[idx]*beta;
622     }
623 }

625 @nogc pure @safe int dim2lin(size_t length)(const ref int[length] ids, const int[] strides) {
626     assert(length == strides.length);
627     import mir.ndslice;
628     import mir.math;
629     return sum(ids.sliced * strides.sliced);

632 /// Reference CPU implementation of Convolution function
633 static struct ConvolutionRefImpl(T, size_t imDims, bool isConv=false, bool isNchw = true) {
634     enum int nbDims = imDims + 2;
635 
636     static void forward(const T[] inputData,
637                         const T[] filterData,
638                         T[] outputData,
639                         T alpha,
640                         T beta,
641                         const int[nbDims] inDims,
642                         const int[nbDims] filDims,
643                         const int[nbDims] outDims,
644                         const int[nbDims] inStride,
645                         const int[nbDims] filStride,
646                         const int[nbDims] outStride,
647                         const int[imDims] stride,
648                         const int[imDims] pad,
649                         const int[imDims] dilation,
650                         )
651     in {
652         // Sanity checks
653         // in     is n x c x h x w
654         // out    is n x k x p x q
655         // filter is k x c x r x s
656         assert(inDims[0] == outDims[0]); // n
657         assert(inDims[1] == filDims[1]); // k
658         assert(outDims[1] == filDims[0]); // c
659     } do {
660         import std.algorithm : reduce;
661 
662         immutable nPixelsOut = outDims[2..$].reduce!"a * b";
663         immutable nPixelsFil = filDims[2..$].reduce!"a * b";
664 
665         // Used to store coordinates
666         int[imDims] filIds, outIds, inIds, tmpIds;
667         // For each image in the output
668         // TODO these three loops can be parallelized without atomic ops
669         foreach (ni; 0 .. outDims[0]) {
670             // For each feature layer of the output
671             foreach (ki; 0 .. outDims[1]) {
672                 immutable outputOffset = ni * outStride[0] + ki * outStride[1] ;
673                 // Loop over all entries of the result
674                 foreach (outId; 0 .. nPixelsOut) {
675                     // Get output pixel ids
676                     lin2dim(outId, outIds, outDims[2..$]) ; // Skip n and k dimensions
677                     // Now we get the coordinates in input space of
678                     // the "top left" corner of the filter: multiply by stride and remove pad
679                     inIds[] = outIds[] * stride[] - pad[];
680                     // We then accumulate
681                     T tmp = 0;
682                     foreach (ci; 0 .. inDims[1]) {
683                         immutable inputOffset = ni * inStride[0] + ci * inStride[1] ;
684                         immutable filterOffset = ki * filStride[0] + ci * filStride[1] ;
685                         foreach (filId; 0 .. nPixelsFil) {
686                             // Get the position of the pixel
687                             lin2dim(filId, filIds, filDims[2..$]) ;
688                             // Compute the corresponding output pixel
689                             // and check wether we are in the padding area on the fly too
690                             // (not that for convolution, we flip the image patch
691                             // (equivalent to flipping the filter patch))
692                             bool inside = true ;
693                             for (int d = 0; d < imDims && inside; d++) {
694                                 if (isConv) {
695                                     tmpIds[d] = inIds[d] + dilation[d] * (filDims[2+d]-1 - filIds[d]);
696                                 } else {
697                                     tmpIds[d] = inIds[d] + dilation[d] * filIds[d];
698                                 }
699                                 // If we are in the padding area: stop and skip computations
700                                 inside &= (tmpIds[d] >= 0 && tmpIds[d] < inDims[2+d]) ;
701                             }
702                             if (inside) {
703                                 immutable actualTmpId = inputOffset + dim2lin(tmpIds, inStride[2..$]);
704                                 immutable actualFilId = filterOffset + dim2lin(filIds, filStride[2..$]);
705                                 tmp += filterData[actualFilId] * inputData [actualTmpId];
706                             }
707                         }
708                     }
709                     // We put the result in the output
710                     immutable actualOutId = outputOffset + dim2lin(outIds, outStride[2..$]);
711                     doEpilog(outputData, actualOutId, alpha*tmp, beta);
712                 }
713             }
714         }
715     }
716 
717     static void backwardData(const T[] weight,
718                              const T[] top_diff,
719                              scope T[] output,
720                              float alpha,
721                              float beta,
722 
723                              const int[nbDims] inDims,
724                              const int[nbDims] filDims,
725                              const int[nbDims] outDims,
726 
727                              const int[nbDims] inStride,
728                              const int[nbDims] filterStride,
729                              const int[nbDims] outStride,
730 
731                              const int[imDims] stride,
732                              const int[imDims] pad,
733                              const int[imDims] dilation)
734     in {
735         // Sanity checks
736         // output is n x c x h x w
737         // diff   is n x k x p x q
738         // filter is k x c x r x s
739         assert(inDims[0] == outDims[0]); // n
740         assert(inDims[1] == filDims[0]); // k
741         assert(outDims[1] == filDims[1]); // c
742     } do {
743         import std.algorithm : map, any, all, reduce;
744         import std.range : iota;
745 
746         // Number of pixels in output
747         immutable nPixelsOut = outDims[2..$].reduce!"a * b";
748         // Number of pixels in filter
749         immutable nPixelsFil = filDims[2..$].reduce!"a * b";
750 
751         int[imDims] outIds, filIds, accIds;
752         // TODO these three loops can be parallelized without atomic ops
753         foreach (ni; 0 .. outDims[0]) {
754             foreach (ci; 0 .. outDims[1]) {
755                 foreach (outIdx; 0 .. nPixelsOut) {
756                     lin2dim(outIdx, outIds, outDims[2..$]);
757                     T val = 0;
758                     // For every diff channel (k)
759                     foreach (ki; 0 .. inDims[1]) {
760                         immutable offsetFilter = ki * filterStride[0] + ci * filterStride[1];
761                         immutable offsetDiff = ni * inStride[0] + ki * inStride[1];
762 
763                         foreach (filIdx; 0 .. nPixelsFil) {
764                             lin2dim(filIdx, filIds, filDims[2..$]);
765 
766                             // Fetch the value in filter and diff, product and accumulate
767                             // So basically, for the convolution,
768                             // we replace r by dim-1-r and s by dim-1-s to "flip" the filter
769                             // We can then just reason in term of correlation
770                             accIds[] = outIds[] + pad[];
771                             if (isConv){
772                                 accIds[] -= (filDims[2..$] - 1 - filIds[]) * dilation[];
773                             } else {
774                                 accIds[] -= filIds[] * dilation[];
775                             }
776                             immutable outtaStride = iota(imDims).map!(i => accIds[i] % stride[i]).any;
777                             if (outtaStride) {
778                                 continue;
779                             }
780                             accIds[] /= stride[];
781 
782                             immutable inBounds = iota(imDims).map!(i => 0 <= accIds[i] && accIds[i] < inDims[i+2]).all;
783                             if (inBounds) {
784                                 immutable filterIdx = offsetFilter + dim2lin(filIds, filterStride[2..$]);
785                                 immutable diffIdx = offsetDiff + dim2lin(accIds, inStride[2..$]);
786                                 val += top_diff[diffIdx] * weight[filterIdx];
787                             }
788                         }
789                         immutable offsetOut = ni * outStride[0] + ci * outStride[1];
790                         doEpilog(output, offsetOut + outIdx, alpha*val, beta);
791                     }
792                 }
793             }
794         }
795     }

797     static void backwardWeight(/*const TensorNdTestDesc_t *tensorInputDesc,*/
798                                const T[] image,
799                                /*const TensorNdTestDesc_t *tensorDiffDesc,*/
800                                const T[] diffData,
801                                /*const ConvNdTestDesc_t *convDesc,*/
802                                /*const TensorNdTestDesc_t *filterOutputDesc,*/
803                                float alpha,
804                                float beta,
805                                scope T[] output,
806 
807                                const int[nbDims] inDims,
808                                const int[nbDims] filDims,
809                                const int[nbDims] diffDims,
810                                const int[nbDims] inStride,
811                                const int[nbDims] filterStride,
812                                const int[nbDims] diffStride,
813                                const int[imDims] stride,
814                                const int[imDims] pad,
815                                const int[imDims] dilation)
816     in {
817         // Some sanity checks
818         // image   is n x c x h x w
819         // diff    is n x k x p x q
820         // filter  is k x c x r x s
821         assert(inDims[0] == diffDims[0]) ;
822         assert(inDims[1] == filDims[1]) ;
823         assert(diffDims[1]  == filDims[0]) ;
824 
825     } do {
826         import std.algorithm : all, sum, map, reduce;
827         import std.range : iota;
828 
829         // Number of pixels in output
830         immutable nPixelsDiff = diffDims[2..$].reduce!"a * b";
831         // Number of pixels in filter
832         immutable nPixelsFil = filDims[2..$].reduce!"a * b";
833 
834         // For every filter pixel (k x c x r x s)
835         int[imDims] filIds, diffIds, accIds;
836         // TODO these three loops can be parallelized without atomic ops
837         foreach (ki; 0 .. filDims[0]){
838             foreach (ci; 0 .. filDims[1]) {
839                 foreach (filIdx; 0 .. nPixelsFil) {
840                     lin2dim(filIdx, filIds, filDims[2..$]);
841                     T val = 0;
842                     // For every image (n)
843                     foreach (ni; 0 .. inDims[0]) { // Sum over the batch
844                         immutable offsetIn = ni * inStride[0] + ci * inStride[1] ;
845                         immutable offsetDiff = ni * diffStride[0] + ki * diffStride[1] ;
846                         // For every pixel in diff
847                         foreach (diffIdx; 0 .. nPixelsDiff) {
848                             lin2dim(diffIdx, diffIds, diffDims[2..$]);
849                             // Fetch the value in image and diff, product and accumulate
850                             accIds[] = diffIds[] * stride[] - pad[];
851 
852                             // Convolution = Correlation with a flipped filter
853                             // So basically, for the convolution, we replace r by dim-1-r and s
854                             // by dim-1-s to "flip" the filter
855                             // We can then just reason in term of correlation
856                             if (isConv){
857                                 accIds[] += (filDims[2..$] - 1 - filIds[]) * dilation[];
858                             } else {
859                                 // The effect of dilation on the gradient is to start the "zone of influence"
860                                 // of a given pixel further into the image, so dilation
861                                 // only produces a shift in x and y
862                                 accIds[] += filIds[] * dilation[];
863                             }
864                             // Image value
865                             immutable inBounds = iota(imDims).map!(i => 0 <= accIds[i] && accIds[i] < inDims[i+2]).all;
866                             if (inBounds) {
867                                 immutable imId = offsetIn + dim2lin(accIds, inStride[2..$]);
868                                 // Diff value
869                                 immutable diffId  = offsetDiff + dim2lin(diffIds, diffStride[2..$]);
870                                 // Prod and accumulate
871                                 val += image[imId] * diffData[diffId];
872                             }
873                         }
874                     }
875                     immutable offsetFilter = ki * filterStride[0] + ci * filterStride[1];
876                     doEpilog(output, offsetFilter + filIdx, alpha*val, beta);
877                 }
878             }
879         }
880     }
