49 /// create new variable with uninitialized array of shape on CPU/CUDA
50 auto uninitVariable(T, alias S = HostStorage, size_t dim)(uint[dim] shape, bool requiresGrad = false) {
51     import std.algorithm : reduce;
52 
53     const length = shape.reduce!"a * b";
54     static if (is(S!T == HostStorage!T)) {
55         auto data = new T[length];
56     }
57     version (grain_cuda) {
58         static if (is(S!T == DeviceStorage!T)) {
59             auto data = CuArray!T(CuPtr!T(length));
60         }
61     }
62     int[dim] strides;
63     strides[dim - 1] = 1;
64     foreach_reverse (i; 0 .. dim - 1) {
65         assert(shape[i + 1] < int.max);
66         strides[i] = cast(int) shape[i + 1] * strides[i + 1];
67     }
68     return Variable!(T, dim, S)(requiresGrad, shape, strides, data);
69 }

113 /// type-erased variable used in BackProp object
114 struct UntypedVariable {
115     import std.variant;
116 
117     bool requiresGrad;
118     size_t dim;
119     // size_t[]
120     uint[] shape;
121     // ptrdiff_t[]
122     int[] strides;
123     TypeInfo elem;
124     Variant data, grad;
125     size_t outPosition = 0;
126     // RefCounted!
127     BackProp bprop;
128 
129     ///
130     this(T, size_t dim, alias Storage)(Variable!(T, dim, Storage) v) {
131         this.elem = typeid(T);
132         this.requiresGrad = v.requiresGrad;
133         this.shape = v.shape.dup;
134         this.strides = v.strides.dup;
135         this.dim = dim;
136         this.data = v.data;
137         this.grad = v.grad;
138     }
139 
140     /// variant.get
141     auto get(T)() {
142         return this.data.get!T;
143     }
144 
145     /// untyped to typed
146     auto to(V : Variable!(T, dim, Storage), T, size_t dim, alias Storage)() {
147         auto d = this.data.get!(Storage!T);
148         return Variable!(T, dim, Storage)(this.requiresGrad,
149                 this.shape[0 .. dim], this.strides[0 .. dim], d);
150     }
151 
152     /// untyped grad to typed
153     auto gradTo(V : Variable!(T, dim, Storage), T, size_t dim, alias Storage)() {
154         auto d = this.data.get!(Storage!T);
155         return Variable!(T, dim, Storage)(this.requiresGrad,
156                 this.shape[0 .. dim], this.strides[0 .. dim], d);
157     }
158 
159     ///
160     string toString() const {
161         import std.format : format;
162 
163         return "UntypedVariable(%s, dim=%d, data=%s, shape=%s, strides=%s)"
164             .format(elem, dim, data, shape, strides);
165     }
166 
167     ///
168     auto gradSlice(V)() if (isVariable!V && isHost!V) {
169         import mir.ndslice.slice : sliced;
170 
171         return grad.get!(typeof(V.init.data)).ptr.sliced(this.shape[0 .. DimensionCount!V]
172                 .castArray!size_t);
173     }
174 
175     ///
176     auto dataSlice(V)() if (isVariable!V && isHost!V) {
177         import mir.ndslice.slice : sliced;
178 
179         return data.get!(typeof(V.init.data)).ptr.sliced(this.shape[0 .. DimensionCount!V]
180                 .castArray!size_t);
181     }
182 }

184 ///
185 auto gradSlice(V)(V v) if (isVariable!V && isHost!V) {
186     import mir.ndslice.slice : sliced;
187 
188     return v.grad.ptr.sliced(v.shape.castArray!size_t);

194 /// stores information for backpropagation
195 struct BackProp {
196     alias Proc = void delegate(UntypedVariable[], UntypedVariable[]);
197     Proc proc;
198     UntypedVariable[] inputs;
199     UntypedVariable[] gradOutputs;
200     size_t nGrad = 0;
201 
202     /// error backward propagation
203     void backward(UntypedVariable* grad = null, size_t pos = 0) {
204         import std.exception : enforce;
205         import std.range : empty;
206 
207         // enforce(!this.inputs.empty, "nothing to backprop");
208         if (this.inputs.empty)
209             return;
210         ++this.nGrad;
211         if (grad is null) {
212             enforce(this.gradOutputs.length == 1, "this variable is not loss");
213         }
214         else {
215             this.gradOutputs[pos] = *grad; // FIXME??
216         }
217         if (grad is null || this.nGrad == this.gradOutputs.length) {
218             proc(this.gradOutputs, this.inputs);
219         }
220 
221         // FIXME: reconsider this maybe
222         // import core.memory : GC;
223         // destroy(gradOutputs);
224         // GC.free(&gradOutputs);
225         // destroy(this);
226         // GC.free(&this);
227     }
228 }

247 struct Variable(T, size_t dim, alias Storage = HostStorage, SliceKind kind = Contiguous) {
248     bool requiresGrad = true;
249     // size_t[dim]
250     uint[dim] shape;
251     // ptrdiff_t[dim]
252     int[dim] strides;
253     Storage!T data;
254     Storage!T grad;
255     // RefCounted!
256     BackProp bprop;
257     enum isHost = is(Storage!T == HostStorage!T);
258     uint offset = 0;
259 
260     ///
261     this(bool requiresGrad, uint[dim] shape, int[dim] strides, Storage!T data) {
262         this.requiresGrad = requiresGrad;
263         this.shape = shape;
264         this.strides = strides;
265         this.data = data;
266         // this.grad.isHost = is(Storage!T == HostStorage!T);
267         // if (this.requiresGrad) { // TODO enable this
268         static if (is(Storage!T == HostStorage!T)) {
269             this.grad = zeros!(Storage!T)(this.data.length);
270         }
271         else version (grain_cuda) {
272             // TODO why is grain.cuda. required?
273             this.grad = grain.cuda.zeros!(CuPtr!T)(this.data.length);
274         }
275         // }
276     }
277 
278     /// get gradient as variable
279     auto gradVariable(bool requiresGrad = false) {
280         return Variable(requiresGrad, this.shape, this.strides, this.grad);
281     }
282 
283     /// detach the computation graph used in backward
284     ref detach() {
285         this.bprop = BackProp();
286         return this;
287     }
288 
289     /// data pointer
290     @property auto ptr() {
291         return this.data.ptr + offset;
292     }
293 
294     /// check data is not null
295     @property bool defined() {
296         return cast(size_t) data.ptr != 0;
297     }
298 
299     /// duplicate (deep copy) variable
300     auto dup() {
301         static if (is(Storage!T == HostStorage!T)) {
302             auto d = new T[data.length];
303             d[] = data[];
304         }
305         else {
306             auto d = CuArray!T(data.dup);
307         }
308         auto y = Variable(this.requiresGrad, this.shape, this.strides, d);
309         return y;
310     }
311 
312     static if (is(Storage!T == HostStorage!T)) {
313         ///
314         auto sliced() {
315             import mir.ndslice; // .slice : Slice, Universal;
316             static if (dim == 0) {
317                 return [this.data[0]].sliced.universal;
318             }
319             else {
320                 return Slice!(T*, dim, Universal)(
321                         this.shape.castArray!size_t,
322                         this.strides.castArray!ptrdiff_t, data.ptr);
323             }
324         }
325 
326         ///
327         auto gradSliced() {
328             import mir.ndslice; // .slice : Slice, Universal;
329             static if (dim == 0) {
330                 return [this.grad[0]].sliced.universal;
331             }
332             else {
333                 return Slice!(T*, dim, Universal)(
334                         this.shape.castArray!size_t,
335                         this.strides.castArray!ptrdiff_t, grad.ptr);
336             }
337         }
338     }
339     else {
340         ///
341         auto sliced() {
342             import mir.ndslice; // .slice : Slice, Universal;
343             static if (dim == 0) {
344                 return Slice!(T*, 1, Universal)([1], [1], cast(T*) data.ptr);
345             }
346             else {
347                 return Slice!(T*, dim, Universal)(
348                         this.shape.castArray!size_t,
349                         this.strides.castArray!ptrdiff_t, cast(T*) data.ptr);
350             }
351         }
352 
353         // TODO gradSliced?
354     }
355 
356     /// computes gradients of creator variables w.r.t. the arg grad
357     void backward(UntypedVariable* grad, size_t pos = 0) {
358         this.bprop.backward(grad, pos);
359     }
360 
361     /// computes gradients of creator variables w.r.t. this variable
362     static if (dim == 0) {
363         void backward() {
364             auto grad = UntypedVariable(1.0f.variable.to!Storage);
365             this.bprop.backward(&grad, 0);
366         }
367     }

370     string toString() const {
371         import std.format : format;
372 
373         return "Variable!(%s, dim=%d, %s)(data=%s, shape=%s, strides=%s)"
374             .format(T.stringof, dim, Storage.stringof, data, shape, strides);

377     /// binary ops: b * this
378     /// TODO implement contiguous with mir.ndslice and cudnnTransformTensor
379     auto opBinary(string op)(Variable!(T, dim, Storage) b) {
380         import grain.chain : opBinaryFunc, reciprocal;
381 
382         static if (op == "+" || op == "*") {
383             return opBinaryFunc!op(this, b);
384         }
385         else static if (op == "-") {
386             return opBinaryFunc!"+"(this, b, 1, -1);
387         }
388         else static if (op == "/") {
389             return opBinaryFunc!"*"(this, reciprocal(b));
390         }
391         else {
392             static assert(false, "unsupported op: " ~ op);
393         }
394     }
395 
396     /// binary ops with primitive scalar value (e.g., float, double)
397     auto opBinary(string op)(T b) {
398         uint[dim] shape;
399         shape[] = 1;
400         auto v = uninitVariable!(T, Storage, dim)(shape, false);
401         static if (is(Storage!T == HostStorage!T)) {
402             import std.algorithm : fill;
403 
404             fill(v.data, b);
405         }
406         else {
407             fill_(v.data, b);
408         }
409         return this.opBinary!op(v);
410     }
411 
412     /// binary ops: this op b
413     auto opBinaryRight(string op)(T b) {
414         static if (op == "+" || op == "*") {
415             return this.opBinary!op(b);
416         }
417         else static if (op == "-") {
418             return this.opBinary!"+"(-b);
419         }
420         else static if (op == "/") {
421             uint[dim] shape;
422             shape[] = 1;
423             auto v = uninitVariable!(T, Storage, dim)(shape, false);
424             static if (is(Storage!T == HostStorage!T)) {
425                 import std.algorithm : fill;
426 
427                 fill(v.data, b);
428             }
429             else {
430                 fill_(v.data, b);
431             }
432             return v.opBinary!op(this);
433         }
434         else {
435             static assert(false, "unsupported op: " ~ op);
436         }
437     }
438 }

543 /// a helper function to create variable object from slice
544 auto variable(Sl)(Sl sl, bool requiresGrad = false) if (isSlice!Sl) {
545     import mir.ndslice : universal, DeepElementType;
546     import std.algorithm : reduce;
547 
548     auto s = sl.universal;
549     alias S = typeof(s);
550     alias E = DeepElementType!S;
551     auto size = s._lengths.reduce!"a * b";
552     auto data = s._iterator[0 .. size];
553     uint[DimensionCount!S] shape;
554     int[DimensionCount!S] strides;
555     static foreach (i; 0 .. DimensionCount!S) {
556         assert(s._lengths[i] < int.max);
557         assert(s._strides[i] < int.max);
558         shape[i] = cast(uint) s.length!i;
559         strides[i] = cast(int) s._strides[i];
560     }
561     return Variable!(E, DimensionCount!S, HostStorage)(requiresGrad, shape, strides, data);
562 }

587 Variable!(T, dim, Dst) to(alias Dst, T, size_t dim, alias Src)(Variable!(T, dim, Src) src) {

630 /// kind of std.algorithm.each for iterating variables inside a chain
631 void iterVariables(alias proc, C)(C* chain, string prefix = "") {
632     import std.traits;
633     import grain.autograd;
634 
635     foreach (name; FieldNameTuple!C) {
636         auto fullName = prefix ~ "." ~ name;
637         auto value = __traits(getMember, chain, name);
638         alias V = typeof(value);
639         static if (isVariable!V) {
640             proc(fullName, value);
641         }
642         else static if (hasMember!(V, "tupleof")) {
643             iterVariables!proc(&value, fullName);
644         }
645     }
646 }
