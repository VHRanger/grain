49 /// sigmoid function
50 struct Sigmoid(T, size_t dim) {
51     import mir.math : exp;
52     import std.math : tanh;
53     import mir.ndslice : sliced, slice, map;
54 
55     Variable!(T, dim, HostStorage) hy;
56 
57     ///
58     auto forward(Variable!(T, dim, HostStorage) x) {
59         enum z = T(0.5);
60         auto ys = x.sliced.map!(a => tanh(a * z) * z + z);
61         auto y = ys.slice.variable(x.requiresGrad);
62         this.hy = y;
63         return y;
64     }
65 
66     ///
67     auto backward(Variable!(T, dim, HostStorage) gy) {
68         auto gx = this.hy.dup;
69         gx.sliced[] *= T(1) - this.hy.sliced;
70         gx.sliced[] *= gy.sliced;
71         return gx;
72     }
73 
74     version (grain_cuda) {
75         // mixin CudnnActivation!(T, dim, CUDNN_ACTIVATION_TANH);
76         enum mode = CUDNN_ACTIVATION_SIGMOID;
77         mixin(CUDNN_ACTIVATION_IMPL_MIXIN);
78     }
79 
80     mixin FunctionCommon;
81 }

110 /// hyperbolic tangent
111 struct Tanh(T, size_t dim) {
112     import std.math : tanh;
113     import mir.ndslice : sliced, slice, map;
114 
115     Variable!(T, dim, HostStorage) hy;
116 
117     ///
118     auto forward(Variable!(T, dim, HostStorage) x) {
119         auto ys = x.sliced.map!tanh;
120         auto y = ys.slice.variable(x.requiresGrad);
121         this.hy = y;
122         return y;
123     }
124 
125     ///
126     auto backward(Variable!(T, dim, HostStorage) gy) {
127         auto gx = this.hy.dup;
128         gx.sliced[] *= gx.sliced; // hy ^^ 2
129         gx.sliced[] = T(1.0) - gx.sliced;
130         gx.sliced[] *= gy.sliced;
131         return gx; // .slice.variable;
132     }
133 
134     version (grain_cuda) {
135         // mixin CudnnActivation!(T, dim, CUDNN_ACTIVATION_TANH);
136         enum mode = CUDNN_ACTIVATION_TANH;
137         mixin(CUDNN_ACTIVATION_IMPL_MIXIN);
138     }
139 
140     mixin FunctionCommon;
141 }

172 /// rectified linear unit nonlinearity (using cuDNN)
173 struct ReLU(T, size_t dim) {
174     mixin FunctionCommon;
175     bool inplace = false;
176     bool useCuDNN = true;
177     Variable!(T, dim, HostStorage) hx;
178 
179     auto forward(Variable!(T, dim, HostStorage) x) {
180         import mir.ndslice : each;
181 
182         // FIXME if train
183         this.hx = x.dup;
184         auto y = this.inplace ? x : x.dup;
185         y.sliced.each!((ref a) {
186             if (a < 0)
187                 a = 0;
188         });
189         return y;
190     }
191 
192     auto backward(Variable!(T, dim, HostStorage) gy) {
193         auto gx = gy.dup;
194         foreach (i; 0 .. gx.data.length) {
195             if (this.hx.data[i] < 0.0)
196                 gx.data[i] = 0.0;
197         }
198         return gx;
199     }
200 
201     // TODO use cudnn
202     version (grain_cuda) {
203         import grain.cudnn;
204 
205         Variable!(T, dim, DeviceStorage) dx, dy;
206 
207         auto forward(Variable!(T, dim, DeviceStorage) x) {
208             // FIXME if train
209             this.dx = x.dup;
210             auto y = this.inplace ? x : x.dup;
211 
212             if (this.useCuDNN) {
213                 activationForward!CUDNN_ACTIVATION_RELU(x, y);
214                 this.dy = y;
215             }
216             else {
217                 import grain.kernel : relu;
218 
219                 auto n = cast(uint) y.data.length; // FIXME use y.nElement
220                 Global.kernel!relu.call(y.data.ptr, n).launch(n);
221             }
222             return y;
223         }
224 
225         auto backward(Variable!(T, dim, DeviceStorage) gy) {
226             auto gx = gy.uninit;
227             if (this.useCuDNN) {
228                 activationBackward!CUDNN_ACTIVATION_RELU(gx, gy, dx, dy);
229             }
230             else {
231                 import grain.kernel : reluGrad;
232 
233                 auto n = cast(uint) gy.data.length;
234                 Global.kernel!reluGrad.call(gx.data.ptr, gy.data.ptr,
235                         this.dx.data.ptr, n).launch(n);
236             }
237             return gx;
238         }
239     }
240 }

373 struct LogSoftmax(T, size_t dim = 2) {
374     // TODO support custom dim to compute softmax over (now only dim=1)
375     mixin FunctionCommon;
376 
377     Variable!(T, dim, HostStorage) hy;
378 
379     auto forward(Variable!(T, dim, HostStorage) x) {
380         import mir.ndslice;
381         import numir;
382 
383         // return slice(x.sliced.alongDim!0.map!(e => e - e.logsumexp)).variable;
384         auto y = x.dup;
385         foreach (i; 0 .. y.shape[0]) {
386             y.sliced[i][] -= x.sliced[i].logsumexp;
387         }
388         // TODO if train
389         this.hy = y;
390         return y;
391     }
392 
393     auto backward(Variable!(T, dim, HostStorage) gy) {
394         import mir.math;
395         import numir;
396         import mir.ndslice;
397 
398         auto gx = gy.dup;
399         auto m = gy.sliced
400             .alongDim!1
401             .map!(sum!"fast");
402         foreach (i; 0 .. gx.shape[0]) {
403             gx.sliced[i][] -= this.hy.sliced[i].map!exp * m[i];
404         }
405         return gx;
406     }
407 
408     version (grain_cuda) {
409         import grain.cudnn;
410 
411         Variable!(T, dim, DeviceStorage) dy;
412 
413         auto forward(Variable!(T, dim, DeviceStorage) x) {
414             auto y = x.dup;
415             softmaxForward!CUDNN_SOFTMAX_LOG(x, y);
416             // TODO if train
417             this.dy = y;
418             return y;
419         }
420 
421         auto backward(Variable!(T, dim, DeviceStorage) gy) {
422             auto gx = gy.dup;
423             softmaxBackward!CUDNN_SOFTMAX_LOG(gx, gy, this.dy);
424             return gx;
425         }
426     }
427 }

549 /// y = 1 / x
550 struct Reciprocal(T, size_t dim) {
551     mixin FunctionCommon;
552 
553     Variable!(T, dim, HostStorage) hy;
554 
555     auto forward(Variable!(T, dim, HostStorage) x) {
556         import mir.ndslice : map, slice;
557 
558         auto y = x.sliced.map!(a => T(1) / a).slice.variable(x.requiresGrad);
559         this.hy = y; // TODO if train
560         return y;
561     }
562 
563     auto backward(Variable!(T, dim, HostStorage) gy) {
564         auto gx = this.hy.dup;
565         gx.sliced[] *= gx.sliced;
566         gx.sliced[] *= -gy.sliced;
567         return gx;
568     }
569 
570     version (grain_cuda) {
571         Variable!(T, dim, DeviceStorage) dy;
572 
573         auto forward(Variable!(T, dim, DeviceStorage) x) {
574             import grain.kernel : reciprocal;
575 
576             auto y = x.dup;
577             unaryFunc!reciprocal(y);
578             this.dy = y; // TODO if train
579             return y;
580         }
581 
582         auto backward(Variable!(T, dim, DeviceStorage) gy) {
583             import grain.cudnn : tensorOp, CUDNN_OP_TENSOR_MUL;
584 
585             auto gx = this.dy.dup;
586             tensorOp!CUDNN_OP_TENSOR_MUL(gx, gx, this.dy, T(-1));
587             tensorOp!CUDNN_OP_TENSOR_MUL(gx, gx, gy);
588             return gx;
589         }
590     }
591 }

1277

1291 auto unpad(size_t d = 0, S, size_t N)(S s, size_t[N] lengths...)
1292         if (isSlice!S && DimensionCount!S == N) {
1293     import mir.ndslice;
1294 
1295     static if (d == N) {
1296         return s;
1297     }
1298     else {
1299         immutable p = lengths[d];
1300         auto s_ = s.swapped!(0, d)[p .. $ - p].swapped!(0, d);
1301         return unpad!(d + 1)(s_, lengths);
1302     }

1313 void sumNdStrided(size_t d = 0, S, D, size_t dim)(S src, D dst, ptrdiff_t[dim] strides...)
1314         if (isSlice!S && isSlice!D && DimensionCount!S == DimensionCount!D) {
1315     static if (d == 0) {
1316         static assert(dim == DimensionCount!S);
1317     }
1318     foreach (i; 0 .. src.length!0) {
1319         immutable j = i * strides[d];
1320         foreach (k; i .. i + strides[d]) {
1321             if (j >= dst.length!0 || k >= src.length!0) {
1322                 break;
1323             }
1324             static if (d + 1 == dim) {
1325                 dst[j] += src[k];
1326             }
1327             else {
1328                 sumNdStrided!(d + 1)(src[k], dst[j], strides);
1329             }
1330         }
1331     }

1397         w[2 .. $] = windowA.castArray!size_t[];

1423 struct Pool(bool isMax, T, size_t poolDims) {
1424     static assert(poolDims > 0);
1425     enum dim = poolDims + 2;
1426     int[poolDims] window, pad, stride;
1427 
1428     Variable!(T, dim, HostStorage) hx, hy;
1429 
1430     static if (isMax) {
1431         // import std.algorithm : max;
1432         import mir.ndslice;
1433 
1434         static auto pool(S)(S s) if (isSlice!S) {
1435             return maxPos(s).first;
1436         }
1437 
1438         alias CpuImpl = PoolRefImpl!pool;
1439     }
1440 
1441     auto forward(Variable!(T, dim, HostStorage) x) {
1442         auto y = CpuImpl.forward(x, this.window, this.pad, this.stride);
1443         // TODO if train
1444         this.hx = x;
1445         this.hy = y;
1446         return y;
1447     }
1448 
1449     auto backward(Variable!(T, dim, HostStorage) gy) {
1450         auto gx = this.hx.uninit;
1451         CpuImpl.backward(gx, this.hx, gy, this.hy, this.window, this.pad, this
1452                 .stride);
1453         return gx;
1454     }
1455 
1456     version (grain_cuda) {
1457         Variable!(T, dim, DeviceStorage) dx, dy;
1458 
1459         import grain.cudnn;
1460 
1461         auto forward(Variable!(T, dim, DeviceStorage) x) {
1462             auto y = grain.cudnn.poolForward(x, this.window, this.pad, this
1463                     .stride);
1464             // TODO if train
1465             this.dx = x;
1466             this.dy = y;
1467             return y;
1468         }
1469 
1470         auto backward(Variable!(T, dim, DeviceStorage) gy) {
1471             auto gx = this.dx.uninit;
1472             grain.cudnn.poolBackward(gx, this.dx, gy, this.dy, this.window, this
1473                     .pad, this.stride);
1474             return gx;
1475         }
1476     }
1477 }
