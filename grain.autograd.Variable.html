Examples

test opBinary(string op)(Variable ...)

1 import mir.ndslice;
2 import numir;
3 import std.stdio;
4 
5 static foreach (op; ["+", "*", "-", "/"]) {
6     {
7         auto a = uniform!float(3, 2).slice.variable(true);
8         auto b = uniform!float(3, 2).slice.variable(true);
9         // this is equivalent to `a + b` if op == "+"
10         auto c = a.opBinary!op(b);
11         // this is equivalent to `a.sliced.slice + b.sliced.slice` if op == "+"
12         auto e = a.sliced.slice.opBinary!op(b.sliced.slice);
13         assert(approxEqual(c.sliced, e));
14 
15         auto gc = uniform!float(3, 2).slice.variable(true);
16         auto ugc = UntypedVariable(gc);
17         c.backward(&ugc);
18 
19         version (grain_cuda) {
20             auto da = a.to!DeviceStorage;
21             auto db = b.to!DeviceStorage;
22             auto dc = da.opBinary!op(db);
23             assert(approxEqual(dc.to!HostStorage.sliced, c.sliced));
24 
25             import grain.cuda : zero_;
26 
27             da.grad.zero_();
28             db.grad.zero_();
29             auto dugc = UntypedVariable(gc.to!DeviceStorage);
30             dc.backward(&dugc);
31             assert(approxEqual(da.to!HostStorage.gradSliced, a.gradSliced));
32         }

test multiple addition

1 grain.autograd.backprop = true;
2 auto x = [1f, 2f].variable(true);
3 auto y = x + x; // x = 2 x
4 auto z = y + y; // x = 4 x
5 auto g = [0f, 1f].variable;
6 auto u = UntypedVariable(g);
7 z.backward(&u);
8 assert(x.gradSliced == [0f, 4f]);
